# 1-my_list.txt

=======================
How to Use 1-my_list.py
=======================

This module defines a class ``MyList`` that inherits from ``list``.

Instantiation
=============

>>> MyList = __import__('1-my_list').MyList
>>> my_list = MyList()
>>> type(my_list)
<class '1-my_list.MyList'>

>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]

>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

>>> print(my_list)
[1, 4, 2, 3, 5]

>>> del my_list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]
>>> my_list.append(-8)
>>> my_list.append(-1)
>>> my_list.append(-2)
>>> my_list.append(-7)
>>> my_list.append(-4)
>>> my_list.print_sorted()
[-8, -7, -4, -2, -1]

my_list = MyList()
>>> my_list.append("Alx")
>>> my_list.append("Holberton")
>>> my_list.append("is")
>>> my_list.append("Cool")
>>> print(my_list)
['Alx', 'Holberton', 'is', 'Cool']
>>> my_list.print_sorted()
['Alx', 'Cool', 'Holberton', 'is']
